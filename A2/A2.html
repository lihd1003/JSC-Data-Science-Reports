def clean(row):
    # clean categories
    error_words = ['& Probates', 'Wine & Spirits', 'Trusts', 'Mags',
                   'Music & Video', 'Vintage & Consignment']
    error_dicts = {'Wills': 'Wills, Trusts, & Probates',
                   'Beer': 'Beer, Wine & Spirits',
                   'Books': 'Books, Mags, Music & Video',
                   'Dry Cleaning & Laundry': 'Dry Cleaning',
                   'Flowers': 'Flowers & Gifts',
                   'Golf Equipment Shops': 'Golf Equipment',
                   'Used': 'Used, Vintage & Consignment'}
    row['categories'] = row['categories'].split(", ") if row['categories'] else []
    i = 0
    while i < len(row['categories']):
        if row['categories'][i] in error_dicts.keys():
            row['categories'][i] = error_dicts[row['categories'][i]]
        if row['categories'][i] in error_words:
            row['categories'].pop(i)
            i -= 1
        i += 1
    return row

business = business_raw.apply(clean, axis=1)
business['coordinates'] = list(zip(business['longitude'], business['latitude']))

For example, "Thai", "Chinese", and "Japanese" are child categories of "Restaurant". In addition, I found the category listing of Yelp's ALL Category List ref , which confirmed my guess. Therefore, I plot the number of business in each root category (categories without parent category) to examine the frequency of business categories.

# create a category dataframe for reference, 
# the category listing is from Yelp's All Category List
# https://www.yelp.com/developers/documentation/v3/all_category_list
category_dict = {}

with urlopen("https://www.yelp.com/developers/documentation/v3/all_category_list/categories.json") as url:
    categories = json.loads(url.read().decode())
    
for item in categories:
    category_dict[item['alias']] =  {'parents': item['parents'], 'category': item['title'], 'level': -1}

category_dict['ethnicfood'] = {'parents': ['food'], 'category': 'Ethnic Food', 'level': -1}
def update_level(name):
    if category_dict[name]['level'] != -1:
        return
    elif not category_dict[name]['parents']:
        category_dict[name]['level'] = 0
    else:
        max_level = 0
        for p in category_dict[name]['parents']:
            update_level(p)
        for p in category_dict[name]['parents']:
            if max_level < category_dict[p]['level'] + 1:
                max_level = 1 + category_dict[p]['level']
        category_dict[name]['level'] = max_level

for key, value in category_dict.items():
    update_level(key)

# make the category list into a dataframe
category_df = {'category': [], 'parent': [], 'level': []}
for key, value in category_dict.items():
    category_df['category'].append(value['category'])
    category_df['level'].append(value['level'])
    parents = [category_dict[i]['category'] for i in value['parents']]
    category_df['parent'].append(parents)
category_df = pd.DataFrame(category_df)
